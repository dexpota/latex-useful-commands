\documentclass{article}
\usepackage{math-commands}
\begin{document}

% \integral examples
\begin{equation}
  \integral{x^2}{x}
\end{equation}

\begin{equation}
  \integral[s]{x^2}{x}
\end{equation}

\begin{equation}
  \integral*[s]{x^2}{x}
\end{equation}

\begin{equation}
  \integral[0][1]{x^2}{x}
\end{equation}

\begin{equation}
  \integral*[0][1]{x^2}{x}
\end{equation}

% \iintegral examples
\begin{equation}
  \iintegral{xy}{x}{y}
\end{equation}

\begin{equation}
  \iintegral[A]{xy}{x}{y}
\end{equation}

\begin{equation}
  \iintegral*[A]{xy}{x}{y}
\end{equation}

% vector functions
\begin{equation}
  \f{dot}{\vec u, \vec v}
\end{equation}

\begin{equation}
  \fdot{u}{v}
\end{equation}

\begin{equation}
  \vdot{u}{v}
\end{equation}

\begin{equation}
  \fcross{u}{v}
\end{equation}

\begin{equation}
  \cross{u}{v}
\end{equation}

\begin{equation}
  \abs{u}
\end{equation}

\end{document}
